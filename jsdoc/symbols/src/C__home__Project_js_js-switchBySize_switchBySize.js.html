 16 /**
 17  * switchBySize.js version 1.0

 23 $.fn.extend({
 24 	/**
 25 	 * Switch CSS by window size interactively.
 26 	 * @function
 27 	 * @param {array} config Collection of rules of switching style by window size, containing formula.
 28 	 * @example $(element).switchBySize([
 29 	 * 	{rule:">=960", css:"style.css"},
 30 	 * 	{rule:"<960", css:"style-narrow.css"}
 31 	 * ]);
 32 	 */
 33 	switchBySize:function( config ){
 34 		var styleobj = this,
 35 			styledir = styleobj.attr("href").replace(/[^\/]+?$/,""),
 36 			switchStyle = function(e){
 37 				var i = config.length,
 38 					w = String( $(e.target).width()),
 39 					c;
 40 				while(i--){
 41 					c = config[ config.length - i - 1 ];
 42 					if( eval( w + c.rule ) ){
 43 						styleobj.attr( "href", styledir + c.css );
 44 						break;
 45 					}
 46 				}
 47 			};
 48 		$(window).bind("resize", switchStyle);
 49 		$(window).trigger("resize");
 50 	}
 51 });
